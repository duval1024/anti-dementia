func isValid(s string) bool {
    m := map[byte]byte {
        ')':'(',
        ']':'[',
        '}':'{',
    }

    stack := []byte{}
    for _, c := range []byte(s) {
        if len(stack) > 0 && stack[len(stack)-1] == m[c] {
            stack = stack[:len(stack) - 1]
        } else {
            stack = append(stack, c)
        }
    }

    return len(stack) == 0
}
